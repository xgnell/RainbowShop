------- Create tables -----

-- Create admins
CREATE TABLE admins (
	id INT AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    gender BOOLEAN NOT NULL,
    birth DATE,
    phone VARCHAR(15) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    passwd VARCHAR(100) NOT NULL,
    rank INT NOT NULL,
    PRIMARY KEY(id)
);

-- Create customers
CREATE TABLE customers (
	id INT AUTO_INCREMENT,
    name VARCHAR(5) NOT NULL,
    gender BOOLEAN NOT NULL,
    birth DATE,
    phone VARCHAR(15) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    passwd VARCHAR(100) NOT NULL,
    address VARCHAR(100),
    PRIMARY KEY (id)
);

-- Create item_types
CREATE TABLE item_types (
	id INT AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    PRIMARY KEY (id)
);

-- Create items
CREATE TABLE items (
	id INT AUTO_INCREMENT NOT null,
    name VARCHAR(50) NOT NULL,
    picture VARCHAR(100),
    price INT NOT NULL,
    description VARCHAR(500),
    id_type INT NOT NULL,
    --material VARCHAR(50),
    color VARCHAR(10),
    PRIMARY KEY (id),
    FOREIGN KEY (id_type) REFERENCES item_types (id)
);

-- Create item_meta_data
CREATE TABLE item_meta_data (
	id_item INT NOT NULL,
    --color VARCHAR(30) NOT NULL,
    size VARCHAR(30) NOT NULL,
    amount INT NOT NULL,
    PRIMARY KEY (id_item, color, size)
);

-- Create item_colors
CREATE TABLE item_colors (
    color VARCHAR(20),
    PRIMARY KEY (color)
);

-- Create item_sizes
CREATE TABLE item_sizes (
	size VARCHAR(20),
    PRIMARY KEY (size)
);
