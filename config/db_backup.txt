-- Create admins
CREATE TABLE admins (
	id INT AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    gender BOOLEAN NOT NULL,
    birth DATE,
    phone VARCHAR(15) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    passwd VARCHAR(100) NOT NULL,
    id_rank INT NOT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (id_rank) REFERENCES admin_ranks (id)
);

-- Create admin_rank
CREATE TABLE admin_ranks (
    id INT AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    level INT NOT NULL
);

-- Create customers
CREATE TABLE customers (
	id INT AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    gender BOOLEAN NOT NULL,
    birth DATE,
    phone VARCHAR(15) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    passwd VARCHAR(100) NOT NULL,
    address VARCHAR(100),
    PRIMARY KEY (id)
);

-- Create item_types
CREATE TABLE item_types (
	id INT AUTO_INCREMENT,
    type VARCHAR(50) NOT NULL,
    PRIMARY KEY (id)
);

-- Create items
CREATE TABLE items (
	id INT AUTO_INCREMENT NOT null,
    name VARCHAR(50) NOT NULL,
    picture VARCHAR(100),
    price INT NOT NULL,
    description VARCHAR(500),
    id_type INT NOT NULL,
    id_color INT NOT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (id_type) REFERENCES item_types (id),
    FOREIGN KEY (id_color) REFERENCES item_colors (id)
);

-- Create item_details
CREATE TABLE item_details (
	id_item INT NOT NULL,
    id_size INT NOT NULL,
    amount INT NOT NULL,
    PRIMARY KEY (id_item, id_size),
    FOREIGN KEY (id_item) REFERENCES items (id),
    FOREIGN KEY (id_size) REFERENCES item_sizes (id)
);

-- Create item_colors
CREATE TABLE item_colors (
    id INT AUTO_INCREMENT,
    color VARCHAR(20) NOT NULL,
    PRIMARY KEY (id)
);

-- Create item_sizes
CREATE TABLE item_sizes (
    id INT AUTO_INCREMENT,
	size VARCHAR(20) NOT NULL,
    PRIMARY KEY (id)
);
